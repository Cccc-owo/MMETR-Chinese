      define_ruler = {
        name = "OS" 	# Xie
        dynasty = "¿R" 	# Liu
        claim = 95
        adm = 5
        dip = 5
        mil = 4
        age = 20
      }
    }
    if = {
      limit = {
        is_emperor_of_china = no
        NOT = {
          has_reform = chinese_claimant_empire_reform
        }
      }
      add_government_reform = chinese_claimant_empire_reform

    if = {
      limit = {
        NOT = {
          government_rank = 3
        }
      }
      set_government_rank = 3
    }

    }
