# Grimoald's Coup
country_event = {
  id = rise_of_the_maior_palatii.7
  title = "rise_of_the_maior_palatii.7.t"
  desc = "rise_of_the_maior_palatii.7.desc"
  picture = CIVIL_WAR_eventPicture
  is_triggered_only = yes
  fire_only_once = yes
  trigger = {
    had_country_flag = {
      flag = rise_of_the_maior_palatii_happened
      days = 365
    }
  }
  immediate = {
    hidden_effect = {
      if = {
        limit = {
          any_owned_province = {
            is_core = ROOT
            is_capital = no
          }
        }
        random_owned_province = {
          limit = {
            is_core = ROOT
            is_capital = no
          }
          spawn_rebels = {
            type = pretender_rebels
            religion = ROOT
            size = 3
            leader = "<hÌ‘«ƒ?–S·V" 	# Grimoald
            leader_dynasty = " Rm—g" 	# Carolingian
            win = yes
          }
          add_local_autonomy = 25
        }
      }
      else = {
        capital_scope = {
          spawn_rebels = {
            type = pretender_rebels
            religion = ROOT
            size = 3
            leader = "<hÌ‘«ƒ?–S·V" 	# Grimoald
            leader_dynasty = " Rm—g" 	# Carolingian
          }
        }
      }
      set_country_flag = grimoalds_coup_started_flag
    }
  }
  option = {
    name = "rise_of_the_maior_palatii.7.a"
    add_manpower = -1
    add_prestige = -10
    add_estate_loyalty = {
      estate = estate_nobles
      loyalty = -5
    }
  }
}
