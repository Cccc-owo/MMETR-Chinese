	is_triggered_only = yes

		name = "mmetr_ITA_Odoacer_mission_events.2.b"

		}

			add_stability = 2		#makes up for -3 stability of conversion effect!
		}
		
		add_country_modifier = {
			name = "me_ita_new_faith"
			duration = 7300
		}
	}
